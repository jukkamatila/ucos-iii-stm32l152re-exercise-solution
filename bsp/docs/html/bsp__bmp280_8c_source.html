793 static int8_t conf_sensor(uint8_t mode, const struct bmp280_config *conf,
794  struct bmp280_dev *dev)
795 {

797  uint8_t temp[2] = {0, 0};
798  uint8_t reg_addr[2] = {BMP280_CTRL_MEAS_ADDR, BMP280_CONFIG_ADDR};
799

802  {
803  rslt = bmp280_get_regs(BMP280_CTRL_MEAS_ADDR, temp, 2, dev);
804  if (rslt == BMP280_OK)
805  {
806  /* Here the intention is to put the device to sleep
807  * within the shortest period of time
808  */

814  temp[0] =

816  temp[1] =
817  BMP280_SET_BITS(temp[1], BMP280_STANDBY_DURN, conf->odr);
818  temp[1] = BMP280_SET_BITS(temp[1], BMP280_FILTER, conf->filter);
819  temp[1] = BMP280_SET_BITS_POS_0(temp[1], BMP280_SPI3_ENABLE,
820  conf->spi3w_en);

822  if (rslt == BMP280_OK)
823  {
824  dev->conf = *conf;

826  {
827  /* Write only the power mode register in a separate
828  * write */

832  }
